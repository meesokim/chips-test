fips_ide_group(examples/z1013)
fips_begin_app(z1013 windowed)
    fips_vs_warning_level(3)
    fips_files(z1013.c)
    if (FIPS_IOS)
        fips_files(ios-info.plist)
    endif()
    fips_deps(roms common)
fips_end_app()
fips_begin_app(z1013-ui windowed)
    fips_vs_warning_level(3)
    fips_files(z1013.c z1013-ui.cc)
    if (FIPS_IOS)
        fips_files(ios-info.plist)
    endif()
    fips_deps(roms common ui)
fips_end_app()
target_compile_definitions(z1013-ui PRIVATE CHIPS_USE_UI)
